<script>
let _results = [];
let _passed = 0, _failed = 0, _skipped = 0;
const startMs = Date.now();

function _addRow(name, status, detail, ms) {
  const div = document.createElement('div');
  div.className = `test-row ${status}`;
  div.innerHTML = `
    <span class="test-badge badge-${status==='pass'?'pass':status==='fail'?'fail':'run'}">${status.toUpperCase()}</span>
    <div style="flex:1">
      <div class="test-name">${name}</div>
      ${detail ? `<div class="test-detail">${detail}</div>` : ''}
    </div>
    ${ms !== undefined ? `<span class="timer">${ms}ms</span>` : ''}
  `;
  document.getElementById('results').appendChild(div);
  document.getElementById('results').scrollTop = 999999;
}

function _sectionHead(label) {
  const div = document.createElement('div');
  div.className = 'section-head';
  div.textContent = '── ' + label;
  document.getElementById('results').appendChild(div);
}

async function runAllTests() {
  document.getElementById('runBtn').disabled = true;
  document.getElementById('results').innerHTML = '';
  document.getElementById('summary').style.display = 'flex';
  _results = []; _passed = 0; _failed = 0; _skipped = 0;
  const t0 = Date.now();

  const suites = window.TSA_TESTS || [];
  for (const suite of suites) {
    _sectionHead(suite.name);
    for (const test of suite.tests) {
      const t1 = Date.now();
      try {
        const result = await test.fn();
        const ms = Date.now() - t1;
        if (result && result.skip) {
          _skipped++;
          _addRow(test.name, 'running', result.reason || 'Skipped', ms);
        } else if (result && result.pass) {
          _passed++;
          _addRow(test.name, 'pass', result.message || '', ms);
        } else {
          _failed++;
          _addRow(test.name, 'fail', result?.message || 'Test returned falsy', ms);
        }
        _results.push({ suite: suite.name, name: test.name, status: result?.skip ? 'skip' : result?.pass ? 'pass' : 'fail', message: result?.message, ms });
      } catch(e) {
        const ms = Date.now() - t1;
        _failed++;
        _addRow(test.name, 'fail', 'Exception: ' + e.message, ms);
        _results.push({ suite: suite.name, name: test.name, status: 'fail', message: 'Exception: ' + e.message, ms });
      }
      // Update counts
      document.getElementById('passCount').textContent = _passed;
      document.getElementById('failCount').textContent = _failed;
      document.getElementById('skipCount').textContent = _skipped;
      document.getElementById('timeCount').textContent = (Date.now()-t0) + 'ms';
      // Breathe between tests
      await new Promise(r => setTimeout(r, 10));
    }
  }

  document.getElementById('runBtn').disabled = false;
  document.getElementById('dlBtn').style.display = 'inline-block';
}

function downloadReport() {
  const report = {
    generated: new Date().toISOString(),
    passed: _passed, failed: _failed, skipped: _skipped,
    total: _results.length,
    results: _results,
  };
  const blob = new Blob([JSON.stringify(report, null, 2)], { type: 'application/json' });
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  a.href = url; a.download = `ts-test-report-${Date.now()}.json`;
  a.click(); URL.revokeObjectURL(url);
}
